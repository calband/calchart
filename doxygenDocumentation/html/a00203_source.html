30 // version 3.1.0 to current
31 // show = START , SHOW ;
32 // START = INGL_INGL , INGL_GURK ;
33 // SHOW = INGL_SHOW , SIZE , { LABEL } , { DESCRIPTION } , { SHEET } , SHOW_END ;
34 // SIZE = INGL_SIZE , BigEndianInt32(4) , BigEndianInt32( number of marchers ) ;
35 // LABEL = INGL_LABL , BigEndianInt32(Sizeof(LABEL_DATA)) , LABEL_DATA ;
36 // LABEL_DATA = { Null-terminated char* } ;
37 // DESCRIPTION = INGL_DESC , BigEndianInt32(Sizeof(DESCRIPTION_DATA)) , DESCRIPTION_DATA ;
38 // DESCRIPTION_DATA = Null-terminated char* ;
39 // SHEET = INGL_GURK , INGL_SHET , NAME , DURATION , POSITION , [ REF_POSITION ] , [ POINT_SYMBOL ] , [ POINT_CONT_INDEX ] , [ POINT_LABEL_FLIP ] , { CONTINUITY } , SHEET_END ;
40 // NAME = INGL_NAME , BigEndianInt32(Sizeof(NAME_DATA)) , NAME_DATA ;
41 // NAME_DATA = Null-terminated char* ;
42 // DURATION = INGL_DURA , BigEndianInt32(4) , BigEndianInt32(number of beats) ;
43 // POSITION = INGL_POS , BigEndianInt32(Sizeof(POSITION_DATA)) , POSITION_DATA ;
44 // POSITION_DATA = { BigEndianInt16( x ) , BigEndianInt16( y ) } ;
45 // REF_POSITION = INGL_REFP , BigEndianInt32(Sizeof(REF_POSITION_DATA)) , REF_POSITION_DATA ;
46 // REF_POSITION_DATA = BigEndianInt16( which reference point ) , { BigEndianInt16( x ) , BigEndianInt16( y ) } ;

48 // POINT_SYMBOL_DATA = { BigEndianInt8( which symbol type ) } ;

50 // POINT_CONT_INDEX_DATA = { BigEndianInt8( which continuity index ) } ;
51 // POINT_LABEL_FLIP = INGL_LABL , BigEndianInt32(Sizeof(POINT_LABEL_FLIP_DATA)) , POINT_LABEL_FLIP_DATA ;
52 // POINT_LABEL_FLIP_DATA = { BigEndianInt8( label flipped ) } ;
53 // CONTINUITY = INGL_CONT , BigEndianInt32(Sizeof(CONTINUITY_DATA)) , CONTINUITY_DATA ;
54 // CONTINUITY_DATA = CONTINUITY_INDEX , CONTINUITY_NAME , CONTINUITY_TEXT ;
55 // CONTINUITY_INDEX = BigEndianInt8( index );
56 // CONTINUITY_NAME = Null-terminated char* ;
57 // CONTINUITY_TEXT = Null-terminated char* ;
58 // SHEET_END = INGL_END , INGL_SHET ;
59 // SHOW_END = INGL_END , INGL_SHOW ;
60 // INGL_INGL = 'I','N','G','L' ;
61 // INGL_GURK = 'G','U','R','K' ;

63 // INGL_SHET = 'S','H','E','T' ;
64 // INGL_SIZE = 'S','I','Z','E' ;
65 // INGL_LABL = 'L','A','B','L' ;
66 // INGL_MODE = 'M','O','D','E' ;
67 // INGL_DESC = 'D','E','S','C' ;
68 // INGL_NAME = 'N','A','M','E' ;
69 // INGL_DURA = 'D','U','R','A' ;
70 // INGL_POS = 'P','O','S',' ' ;
71 // INGL_SYMB = 'S','Y','M','B' ;
72 // INGL_TYPE = 'T','Y','P','E' ;
73 // INGL_REFP = 'R','E','F','P' ;
74 // INGL_CONT = 'C','O','N','T' ;
75 // INGL_PCNT = 'P','C','N','T' ;
76 // INGL_END = 'E','N','D',' ' ;
77 // Some assumptions:
78 // strings are ascii 8-bit chars.
79 // Error if LABEL_DATA does not contain N null terminated strings, where N == number of marchers
80 // Error if POSITION_DATA does not contain N*2 values, where N == number of marchers
81 // Error if REF_POSITION_DATA does not contain N*2 + 1 values, where N == number of marchers
82 // Error if POINT_SYMBOL_DATA does not contain N values, where N == number of marchers
83 // Error if POINT_CONT_INDEX_DATA does not contain N values, where N == number of marchers
84 // Error if POINT_LABEL_FLIP_DATA does not contain N values, where N == number of marchers
85 // If REF_POSITION is not supplied, all reference points are assumed to be set to the point value
86 // If POINT_SYMBOL is not supplied, all points assumed to be symbol 0
87 // If POINT_CONT_INDEX is not supplied, all points assumed to be index 0
88 // If POINT_LABEL_FLIP is not supplied, all points assumed to be not flipped

161 uint16_t get_big_word(const T* ptr)
162 {
163  return ((ptr[0] & 0xFF) << 8) | ((ptr[1] & 0xFF));
164 }

166 template <typename T>
167 uint32_t get_big_long(const T* ptr)
168 {
169  return ((ptr[0] & 0xFF) << 24) | ((ptr[1] & 0xFF) << 16) | ((ptr[2] & 0xFF) << 8) | ((ptr[3] & 0xFF));
170 }

239 template <typename T>
240 inline uint32_t ReadGurkSymbolAndGetVersion(T& stream, uint32_t inname)
241 {
242  uint32_t name = ReadLong(stream);
243  if (inname == name)
244  {
245  // if we have an exact match, this is version 0
246  return 0x0;
247  }
248  if ((((name>>24)&0xFF) == ((inname>>24)&0xFF)) && (((name>>16)&0xFF) == ((inname>>16)&0xFF)))
249  {
250  char major_vers = (name>>8) & 0xFF;
251  char minor_vers = (name) & 0xFF;
252  if (isdigit(major_vers) && isdigit(minor_vers))
253  {
254  return ((major_vers - '0')<<8) | (minor_vers - '0');
255  }
256  }
257  throw CC_FileException(inname);
258 }
